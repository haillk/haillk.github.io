//错误
func main() {
	recover()
	panic(1)
}
//错误
func main() {
	defer recover()
	panic(1)
}
//正确
func main() {
	defer func() {
		recover()
	}()
	panic(1)
}
//错误
func main() {
	defer func() {
		defer func(){
			recover()
		}
		
	}()
	panic(1)
}
